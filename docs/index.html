Atom Engine v5

A high-performance, component-oriented template engine for Rust built on Tera.

🚀

Built on Tera

Leverages Tera's robust parsing, caching, and expression evaluation

🧩

Component System

Reusable components with props, slots, and validation

🔄

50+ Filters

String, collection, number, date, HTML, and conditional filters

⚡

High Performance

Async support, parallel rendering, and component caching

Installation

Add Atom Engine to your Cargo.toml:

[dependencies]
atom-engine = "5"

For additional features:

[dependencies]
atom-engine = { version = "5", features = ["parallel", "async"] }
Feature Description
parallel Enable parallel rendering with Rayon
async Enable async template rendering with Tokio
pool-alloc Enable memory pooling for allocations

Quick Start

use atom_engine::Atom;
use serde_json::json;

fn main() {
    let mut engine = Atom::new();
    
    engine.add_template("hello.html", "Hello, {{ name }}!").unwrap();
    
    let result = engine.render("hello.html", &json!({"name": "World"})).unwrap();
    println!("{}", result);  // Hello, World!
}

Template Syntax

Variables

{{ user.name }}
{{ items[0] }}
{{ company.address.city }}
{{ items | length }}

Comments

{# This is a comment #}

Filters

Filters transform values. Chain multiple filters:

{{ name | upper | capitalize }}
{{ text | truncate(length=50, end="...") }}
{{ items | sort | reverse | first }}

Conditionals

{% if user.is_active %}
    Welcome, {{ user.name }}!
{% elif user.is_pending %}
    Please verify your email.
{% else %}
    <a href="/login">Login</a>
{% endif %}

Boolean Operators

{% if user.is_active and user.is_verified %}
    Verified user
{% endif %}

{% if is_admin or is_editor %}
    Can edit content
{% endif %}

{% if not user.is_banned %}
    Not banned
{% endif %}

Loops

{% for item in items %}
    <li>{{ item }}</li>
{% endfor %}

{% for item in items %}
    Index: {{ loop.index }}, Value: {{ item }}
    {% if loop.first %}First!{% endif %}
    {% if loop.last %}Last!{% endif %}
{% endfor %}

Template Inheritance

{% extends "base.html" %}

{% block title %}{{ page_title }}{% endblock %}

{% block content %}
    <h1>{{ article.title }}</h1>
{% endblock %}

Includes

{% include "header.html" %}
{% includeIf "optional.html" %}
{% includeFirst ["primary.html", "fallback.html"] %}

Macros

{% macro button(text, type="primary") %}
<button class="btn btn-{{ type }}">{{ text }}</button>
{% endmacro %}

{{ button("Submit", type="success") }}

String Filters

Filter Description Example
upper Uppercase {{ "hello" | upper }} → HELLO
lower Lowercase {{ "HELLO" | lower }} → hello
capitalize Capitalize first letter {{ "hello" | capitalize }} → Hello
title Title case {{ "hello world" | title }} → Hello World
camel_case camelCase {{ "hello_world" | camel_case }} → helloWorld
pascal_case PascalCase {{ "hello_world" | pascal_case }} → HelloWorld
snake_case snake_case {{ "helloWorld" | snake_case }} → hello_world
kebab_case kebab-case {{ "helloWorld" | kebab_case }} → hello-world
truncate(length, end?) Truncate string {{ "Hello World" | truncate(length=5) }} → Hello...
slugify URL-safe slug {{ "Hello World!" | slugify }} → hello-world
replace(old, new) Replace text {{ "Hi" | replace(old="Hi", new="Hello") }} → Hello
pluralize Pluralize {{ 2 | pluralize }} → s

Helper Directives

@map - Transform Array Elements

{{ users | map(prop="name") | join(", ") }}
{{ users | map(prop="name", transform="upper") }}

@filter - Filter Array Elements

{{ users | filter(key="active", value=true) }}
{{ users | filter(key="age", value=18, op="gte") }}

@each - Iterate with Index

{% for item in items | each(index=true) %}
    {{ item.index }}: {{ item.value }}
{% endfor %}

@reduce - Reduce to Value

{{ numbers | reduce(initial=0) }}
{{ users | reduce(initial=0, prop="age") }}

@flatten - Flatten Arrays

{{ [[1,2], [3,4]] | flatten | join(", ") }}
{# Output: 1,2,3,4 #}

@partition - Split Array

{% set partitioned = users | partition(key="active") %}
Matched: {{ partitioned.matched | length }}
Rest: {{ partitioned.rest | length }}

Advanced Features

Component Caching

engine.enable_component_cache(true);
let result = engine.render("page.html", &ctx).unwrap();
println!("Cache size: {}", engine.component_cache_len());
engine.clear_component_cache();

Async Rendering

#[cfg(feature = "async")]
async fn render() {
    let engine = Atom::new();
    let result = engine.render_async("template.html", &json!({})).await.unwrap();
}

Parallel Rendering

#[cfg(feature = "parallel")]
let results = engine.render_many(&[
    ("t1.html", &json!({})),
    ("t2.html", &json!({})),
]).unwrap();
